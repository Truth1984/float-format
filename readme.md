## float format

example

```js
const ff = require("float-format");
ff(0.2 * 0.2); // 0.04
0.2 * 0.2; // 0.04000000000000001

0.2 * 0.2 == 0.04; // false
ff(0.2 * 0.2) == 0.04; // true
0.2 * 0.2 - 0.04; // 6.938893903907228e-18

ff(0.00015625) == 2 / 12800; // true
ff(2 / 12800) == 2 / 12800; //true
```
